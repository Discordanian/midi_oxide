enum EventType {
    NoteOff,
    NoteOn,
    Other,
}
struct MidiEvent {
    event: EventType,

    key: i8,
    velocity: i8,
    delta_tick: u32,
}

struct MidiNote {
    key: i8,         // 7 bit
    velocity: i8,    // 7 bit
    start_time: i32, // 28 bit (4*7bit)
    duration: i32,   // 28 bit (4*7bit)
}

struct MidiTrack {
    name: String,
    instrument: String,

    events: Vec<MidiEvent>,
    notes: Vec<MidiNote>,

    max_note: i8,
    min_note: i8,
}
